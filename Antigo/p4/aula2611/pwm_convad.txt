#INCLUDE<P16F877A.INC>

 org 0x00
 GOTO INICIO

 ORG 0X08

 REG1 EQU 20H
 REG2 EQU 21H
 CONT EQU 22H

 CH0H EQU 24H
 VAR1 EQU 25H
 VAR2 EQU 26H
 UNI  EQU 27H

 CEN  EQU 29H
 X    EQU 30H

 INICIO

 BCF STATUS,RP1
 MOVLW B'11111111'
 MOVWF TRISA

 MOVWF TRISB
 MOVLW B'11111001'
 MOVWF TRISC
 MOVLW 0X00
 MOVWF TRISD
 MOVWF TRISE

 MOVWF ADCON1 ;HABILITA ENTRADAS ANALOGICAS!
 MOVLW B'10000000'

 MOVLW B'01000001'
 MOVWF ADCON0
 CALL ATRASO_1S
 CALL INICIA_LCD
 MOVLW 0X00
 MOVWF CH0L
 MOVWF CH0H


 ;Habilita PWM
 MOVLW B'00001111' 
MOVWF CCP1CON ;HABILITA PWM1
 MOVWF CCP2CON ;HABILITA PWM2

 BSF STATUS,RP0
 MOVLW 0xFF
 MOVWF PR2 ;RESOLUÇÃO DE 10 BITS
 BCF STATUS,RP0
 BSF T2CON,TMR2ON ;TIMER LIGADO
 MOVLW .0
 MOVWF VAR1
 MOVLW .0
 MOVWF VAR2

 LOOP

 CALL LER_ADC
 CALL ATRASO_1S 

 BTFSS PORTB,3 ;TESTE BOTÃO1
 CALL BOTAO1
 CALL CONVERSAO_BIT_DEC
 GOTO	LOOP

 LER_ADC
 BSF ADCON0,GO
 BTFSC ADCON0,GO
 GOTO $-1
 BSF STATUS,RP0
 MOVF ADRESL,W
 BCF STATUS,RP0
 MOVWF CH0L
 MOVF ADRESH,W
 MOVWF CH0H
 RETURN

DECODER

ADDWF	PCL,F
RETLW	'0'
RETLW	'1'
RETLW	'2'
RETLW	'3'
RETLW	'4'
RETLW	'5'
RETLW	'6'
RETLW	'7'
RETLW	'8'
RETLW	'9'


 INICIA_LCD
 MOVLW 0X38
 CALL ESCREVE_COMANDO
 CALL ATRASO_1MS
 CALL ATRASO_1MS
 CALL ATRASO_1MS
 MOVLW 0X38
 CALL ESCREVE_COMANDO
 MOVLW 0X0C
 CALL ESCREVE_COMANDO
 MOVLW 0X06
 CALL ESCREVE_COMANDO
 MOVLW 0X01
 CALL ESCREVE_COMANDO
 CALL ATRASO_1MS
 RETURN

 BOTAO1
 MOVLW .100
 ADDWF VAR1,F

 BTFSC STATUS,C
 INCF VAR2,F

 CALL ATUALIZA_PWM1

 CALL ATRASO_1S
 BTFSS PORTB,3
 GOTO $-1
 RETURN

 ESCREVE_COMANDO ;ENVIA COMANDO AO LCD
 BCF PORTE,0 ;SINAL DE COMANDO
 NOP
 MOVWF PORTD ;INFORMAÇÃO NA PORTA D DE DAOD  NOP  NOP  BSF PORTE,1 ;PULSO DE ENABLE  NOP  NOP  BCF PORTE,1 CALL ATRASO_1MS ;ATRSO DE 1ms  RETURN

 ESCREVE_DADO ;ESCREVE CARACTERES NO LCD BSF PORTE,0 ;SINAL DE DADO  NOP  MOVWF PORTD ;INFORMAÇÃO NA PORTA D DE DAOD  NOP  NOP  BSF PORTE,1 ;PULSO DE ENABLE  NOP  NOP  BCF PORTE,1 CALL ATRASO_1MS ;ATRSO DE 1ms  RETURN 

ATUALIZA_PWM1 ;ATUALIZA O VALOR DA LARGURA DE PULSO DO PWM.
 BTFSC VAR1,0
 BSF CCP1CON,4
 BTFSS VAR1,0
 BCF CCP1CON,4

 BTFSC VAR1,1
 BSF CCP1CON,5
 BTFSS VAR1,1
 BCF CCP1CON,5

 BTFSC VAR1,2

 BTFSS VAR1,2
 BCF CCPR1L,0

 BTFSS VAR1,3

 BCF CCPR1L,2

 BCF CCPR1L,3

 BSF CCPR1L,4

 BCF CCPR1L,4

 BSF CCPR1L,5
 BTFSS VAR1,7
 BCF CCPR1L,5

 BTFSC VAR2,0
 BSF CCPR1L,6
 BTFSS VAR2,0
 BCF CCPR1L,6

 BTFSC VAR2,1
 BSF CCPR1L,7
 BTFSS VAR2,1
 BCF CCPR1L,7
 RETURN

CONVERSAO_BIT_DEC
 CLRF UNI
 CLRF DEC
 CLRF CEN

 BTFSC CH0H,1

 BTFSS CH0H,0

 BTFSS CH0L,6

 BTFSS CH0L,5
 BCF X,3

 BTFSC CH0L,4
 BSF X,2
 BTFSS CH0L,4
 BSF X,2

 BTFSC CH0L,3
 BSF X,1
 BTFSS CH0L,3
 BCF X,1

 BTFSC CH0L,2
 BSF X,0
 BTFSS CH0L,2
 BCF X,0
  
TESTA
 MOVLW .0
 SUBWF X,W
 BTFSC STATUS,Z
GOTO EXIBE

 DECF X,F
 INCF CEN,F
 INCF CEN,F
 MOVLW .10
 SUBWF CEN,W
 BTFSS STATUS,Z
GOTO TESTA

 CLRF CEN
 INCF DEC,F
 MOVLW .10
 SUBWF DEC,W
 BTFSS STATUS,Z
GOTO TESTA

 CLRF DEC
 INCF UNI,F
GOTO TESTA

 EXIBE
 MOVLW 0X80
 CALL ESCREVE_COMANDO
 MOVF UNI,W
 CALL DECODER
 CALL ESCREVE_DADO
 MOVLW ','

 MOVF DEC,W
 CALL DECODER
 CALL ESCREVE_DADO
 MOVF CEN,W

 CALL ESCREVE_DADO
 RETURN
